    async function callGemini(prompt, systemPrompt) {

        const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`;

        for (let i = 0; i < 5; i++) {

            try {

                const response = await fetch(url, { 

                    method: 'POST', 

                    headers: { 'Content-Type': 'application/json' },

                    body: JSON.stringify({ 

                        contents: [{ parts: [{ text: prompt }] }], 

                        systemInstruction: { parts: [{ text: systemPrompt }] } 

                    }) 

                });

                const result = await response.json();

                return result.candidates?.[0]?.content?.parts?.[0]?.text;

            } catch (e) { await new Promise(res => setTimeout(res, Math.pow(2, i) * 1000)); }

        }

        return "Coaching unavailable right now. Keep moving!";

    }

    function setVersion(p, v) { 

        currentVersions[p] = v;

        const pBtnA = document.getElementById(`v-btn-${p === 'upper' ? 'uA' : 'lA'}`);

        const pBtnB = document.getElementById(`v-btn-${p === 'upper' ? 'uB' : 'lB'}`);

    function showTab(t, b) { 

        document.querySelectorAll('.panel').forEach(p => p.classList.remove('active'));

        document.querySelectorAll('.tab-btn').forEach(btn => btn.classList.remove('active'));

        document.getElementById(t).classList.add('active'); 

        b.classList.add('active');

        if(t === 'vault') renderHistory(); 

        window.scrollTo(0,0);

    }


    function toggleCheck(e) { 

        e.classList.toggle('checked'); 

        e.innerHTML = e.classList.contains('checked') ? '<i data-lucide="check" style="width:14px;height:14px;color:#000"></i>' : ''; 

        lucide.createIcons(); 

    }


    function startTimer(s) {

        const o = document.getElementById('timer-overlay'), c = document.getElementById('timer-count');

        o.style.display = 'flex'; 

        let r = s; 

        c.innerText = r;

        clearInterval(timerInterval);

        timerInterval = setInterval(() => { 

            r--; 

            c.innerText = r; 

            if(r <= 0) { 

                closeTimer(); 

                try {

                    const ctx = new (window.AudioContext || window.webkitAudioContext)();

                    const osc = ctx.createOscillator();

                    osc.connect(ctx.destination);

                    osc.start(); osc.stop(ctx.currentTime + 0.2);

                } catch(e){} 

            } 

        }, 1000);

    }


    function closeTimer() { document.getElementById('timer-overlay').style.display = 'none'; clearInterval(timerInterval); }


    function saveExercise(btn, name) {

        const row = btn.parentElement, w = row.querySelector('.w-input').value, r = row.querySelector('.r-input').value;

        if(!w || !r) return;

        const entry = { name, w, r, date: new Date().toLocaleDateString('en-US', { month: 'short', day: 'numeric' }) };

        let logs = JSON.parse(localStorage.getItem('ff_logs_v8') || '[]'); 

        logs.unshift(entry);

        localStorage.setItem('ff_logs_v8', JSON.stringify(logs));

        btn.innerHTML = '✓'; 

        btn.style.background = 'var(--good)';

        setTimeout(() => { 

            btn.innerHTML = '+'; 

            btn.style.background = 'var(--accent)'; 

            row.querySelector('.w-input').value = ''; 

            row.querySelector('.r-input').value = ''; 

        }, 1200);

    }


    function renderHistory() {

        const logs = JSON.parse(localStorage.getItem('ff_logs_v8') || '[]');

        const container = document.getElementById('history-list');

        if(logs.length === 0) { 

            container.innerHTML = '<div style="text-align:center; padding: 40px; color: var(--muted);">Vault is empty. Log your sets in the Upper/Lower tabs.</div>'; 

            return; 

        }

        container.innerHTML = logs.map(l => `

            <div class="history-item">

                <div>

                    <div class="history-date">${l.date}</div>

                    <div style="font-weight:700;">${l.name}</div>

                </div>

                <div class="history-val">${l.w} lbs × ${l.r}</div>

            </div>

        `).join('');

    }
